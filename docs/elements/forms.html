Validation#

PIN code is valid.
PIN code must have at least 4 characters.
The option is invalid.
Use input pattern attribute to validate the value as well.
<div class="form-element">
    <input class="form-field is-success" id="pin-success" type="text" placeholder="PIN code">
    <div class="form-hint">PIN code is valid.</div>
</div>
<div class="form-element">
    <input class="form-field is-error" id="pin-error" type="text" placeholder="PIN code">
    <div class="form-hint">PIN code must have at least 4 characters.</div>
</div>
<div class="form-element">
    <label class="form-label" for="car-error">Choose an option</label>
    <select class="form-field is-error" id="car-error">
        <option>Volvo</option>
        <option>Saab</option>
        <option>Mercedes</option>
        <option>Audi</option>
    </select>
    <div class="form-hint">The option is invalid.</div>
</div>
<div class="form-element">
    <label class="form-check is-error"> Error radio
        <input type="radio" name="radio">
        <span class="form-mark"></span>
    </label>
</div>
<div class="form-element">
    <label class="form-check is-error"> Error checkbox
        <input type="checkbox" name="checkbox">
        <span class="form-mark"></span>
    </label>
</div>
<div class="form-element">
    <label class="form-label" for="email-pattern">Email</label>
    <input class="form-field" id="email" type="email-pattern" pattern="[a-z0-9._%+-]+@[a-z0-9.-]+.[a-z]{2,14}$" placeholder="Start typing...">
    <div class="form-hint">Use input pattern attribute to validate the value as well.</div>
</div>
