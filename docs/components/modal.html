Show#

You can show modal with the example below

document.addEventListener("DOMContentLoaded", function () {
    [].forEach.call(
        document.querySelectorAll("[data-toggle='modal']"),
        function (el) {
            el.onclick = function () {
                document.getElementById(el.getAttribute("data-target")).classList.add("show");
                return false;
            };
        }
    );
});

Hide#

You can hide modal with the example below

document.addEventListener("DOMContentLoaded", function () {
    [].forEach.call(
        document.querySelectorAll("[data-dismiss='modal'"),
        function (el) {
            el.onclick = function () {
                el.closest(".modal").classList.remove("show");
                return false;
            };
        }
    );
});
