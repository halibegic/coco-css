Dismiss#

Add the close button to make an alert modal dismissible. Thanks to that, your alert modal will disappear only once the user closes it.

Lorem ipsum dolor sit amet, consectetur adipiscing elit.
<div class="alert alert-success" role="alert">
    <div class="d-flex">
        <!-- Icon with class="alert-icon" -->
        <div>Lorem ipsum dolor sit amet, consectetur adipiscing elit.</div>
    </div>
    <button class="close alert-close" data-dismiss="alert">&times;</button>
</div>

document.addEventListener("DOMContentLoaded", function () {
    [].forEach.call(
        document.querySelectorAll("[data-dismiss='alert']"),
        function (el) {
            el.onclick = function () {
                el.closest(".alert").style.display = "none";
                return false;
            };
        }
    );
});
