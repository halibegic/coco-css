document.addEventListener("DOMContentLoaded", function () {
    [].forEach.call(
        document.querySelectorAll("[data-dismiss='alert']"),
        function (el) {
            el.onclick = function () {
                el.closest(".alert").style.display = "none";
                return false;
            };
        }
    );
});
